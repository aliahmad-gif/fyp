// Load recent labels on start window.addEventListener('DOMContentLoaded', () => { const urlParams = new URLSearchParams(window.location.search); const activeTab = urlParams.get('tab'); if (activeTab === 'tailor') { switchTab({ currentTarget: document.querySelector('.tab-btn:nth-child(2)') }, 'tailor-chats'); } else if (activeTab === 'user') { switchTab({ currentTarget: document.querySelector('.tab-btn:nth-child(1)') }, 'user-chats'); } }); function switchTab(evt, tabName) { const panes = document.querySelectorAll('.tab-pane'); panes.forEach(pane => pane.classList.remove('active')); const btns = document.querySelectorAll('.tab-btn'); btns.forEach(btn => btn.classList.remove('active')); document.getElementById(tabName).classList.add('active'); evt.currentTarget.classList.add('active'); } function handleLogout() { if (confirm("Are you sure you want to logout?")) { // Clear local storage localStorage.removeItem('isLoggedIn'); localStorage.removeItem('currentUser'); localStorage.removeItem('userRole'); // Redirect to Main Index window.location.href = "../index.html"; } }
